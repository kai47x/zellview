MODULE TexelEaster2;
(* *)

IMPORT Log, Dialog, Dates,
	DevProfiler;

	PROCEDURE Find*( from, len: INTEGER );
	VAR 
		i: INTEGER;
		str: Dialog.String;
		date: Dates.Date;
	BEGIN
		Log.Ln;
		Log.String( "Ostersonntag im Jahr " ); Log.Int( from );
		Log.String( " bis " ); Log.Int(from+len); Log.Ln;
		
		DevProfiler.SetModuleList("TexelEaster");
		DevProfiler.Start;

		FOR i := from TO from + len DO
			Dates.GetEasterDate( i, date);
			
			Dates.DateToString(date, Dates.plainLong, str); 
			Log.Int(i); Log.Tab;
			Log.String(str); Log.Ln;
		
		END;
		
		DevProfiler.Stop;
			
		
		Log.String( "Ostersonntage gefunden. Frohe Ostern");
	END Find;

	PROCEDURE Do*;
	VAR
		startYear, years: INTEGER;
	BEGIN

		startYear := 1599;
		years := 200;
		
		Find( startYear, years );

	END Do;

END TexelEaster2.
TexelEaster2.Do
